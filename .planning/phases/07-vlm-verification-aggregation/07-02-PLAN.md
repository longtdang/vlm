      Rule-to-annotation-fields mapping (dict[str, list[str]]):
        RULE_ANNOTATION_FIELDS: dict[str, list[str]] = {
            "bbox_localization": ["bbox"],
            "bbox_coverage": ["bbox", "attributes"],
            "clamp_type": ["attributes"],
            "roll_count": ["attributes"],

            "occlusion_state": ["keypoints", "visibility"],

      def build_prompt(template: str, label: str, rule: str, annotation: dict[str, Any]) -> str:
        - Extract rule-relevant fields from annotation using RULE_ANNOTATION_FIELDS[rule] (or all fields if rule not in map)
        - Build annotation_fields dict with only the relevant keys (skip keys with None/empty values when possible but include if explicitly set)
        - annotation_fields_json = json.dumps(annotation_fields, ensure_ascii=False)
        - Return template.format(label=label, rule=rule, annotation_fields_json=annotation_fields_json)
        - On KeyError (template missing placeholder): return template as-is with what formatted successfully
          (use str.format_map with a defaultdict fallback to prevent KeyError)

      def parse_vlm_response(raw: str, rule_name: str) -> VlmRuleResult:
        - stripped = raw.strip()
        - Apply _FENCE_RE.search(stripped); if match, stripped = match.group(1).strip()
        - try json.loads(stripped); on (JSONDecodeError, ValueError) → return VlmRuleResult with invalid_output=True,
          error_probability=None, reason="invalid_output: JSON parse failed", evidence=None
        - ep = parsed.get("error_probability")
        - if not isinstance(ep, (int, float)) or not (0.0 <= float(ep) <= 1.0):
            return VlmRuleResult with invalid_output=True, error_probability=None,
                   reason=f"invalid_output: error_probability={ep!r}", evidence=None
        - return VlmRuleResult(rule_name=rule_name, error_probability=float(ep),
            reason=str(parsed.get("reason", "")), evidence=parsed.get("evidence"),
            invalid_output=False)

      def evaluate_vlm_object(
          *,
          result: ObjectVerificationResult,
          annotation: dict[str, Any],
          crop_image: PILImage.Image,
          adapter: VlmAdapter,
          vlm_config: VlmConfig,
      ) -> VlmObjectResult:
        - rules = vlm_config.rules_for_label(result.label)
        - thresholds = vlm_config.thresholds_for_label(result.label)
        - rule_results: list[VlmRuleResult] = []
        - failure_reason: str | None = None
        - adapter_model = vlm_config.model_name
        - For each rule in rules:
            per_label_template = vlm_config.prompt_for_label_rule(result.label, rule)
            template = per_label_template if per_label_template is not None else vlm_config.default_prompt_template
            prompt = build_prompt(template, result.label, rule, annotation)
            try:
                raw_text = _call_with_timeout(
                    adapter.generate_text,
                    timeout_seconds=vlm_config.generation.timeout_seconds,
                    image=crop_image,
                    prompt=prompt,
                )
                rule_result = parse_vlm_response(raw_text, rule)
                rule_results.append(rule_result)
            except TimeoutError as exc:
                failure_reason = f"adapter_timeout:{exc}"
                return VlmObjectResult(
                    sample_id=result.sample_id, object_id=result.object_id,
                    label=result.label, vlm_status=VlmVerdict.REVIEW,
                    object_risk=None, rule_results=rule_results,
                    adapter_model=adapter_model, failure_reason=failure_reason,
                    crop_path=result.crop_path,
                )
            except Exception as exc:
                failure_reason = f"adapter_error:{type(exc).__name__}:{exc}"
                # Per D-03/D-04: single model, fail → REVIEW; abort remaining rules
                return VlmObjectResult(
                    sample_id=result.sample_id, object_id=result.object_id,
                    label=result.label, vlm_status=VlmVerdict.REVIEW,
                    object_risk=None, rule_results=rule_results,
                    adapter_model=adapter_model, failure_reason=failure_reason,
                    crop_path=result.crop_path,
                )
        - valid_results = [r for r in rule_results if not r.invalid_output]
        - if not valid_results:
            return VlmObjectResult(..., vlm_status=VlmVerdict.REVIEW, object_risk=None,
                failure_reason="all_invalid_output", ...)
        - object_risk = max(r.error_probability for r in valid_results)
        - vlm_status: VlmVerdict
          if object_risk <= thresholds.pass_below: vlm_status = VlmVerdict.PASS
          elif object_risk <= thresholds.review_below: vlm_status = VlmVerdict.REVIEW
          else: vlm_status = VlmVerdict.FAIL
        - return VlmObjectResult(
            sample_id=result.sample_id, object_id=result.object_id,
            label=result.label, vlm_status=vlm_status, object_risk=object_risk,
            rule_results=rule_results, adapter_model=adapter_model,
            failure_reason=None, crop_path=result.crop_path,
          )
