    Create src/fiftyone_pose_importer/verification/vlm_client.py:
      - "from __future__ import annotations" header
      - from __future__ import annotations
      - from typing import Protocol, runtime_checkable
      - from PIL import Image as PILImage  (always available via fiftyone transitive dep)

      - @runtime_checkable class VlmAdapter(Protocol):
          def generate_text(self, image: PILImage.Image, prompt: str) -> str: ...

      - class FiftyOneZooAdapter:
          """Thin wrapper around FiftyOne zoo Qwen3-VL model for raw text generation.
          Loads the model lazily on first call. Not thread-safe — single sequential pipeline use only.
          Per D-01: FiftyOne model-zoo only. No external HTTP endpoint in this implementation.
          """
          def __init__(self, model_name: str, max_new_tokens: int = 256) -> None:
              self._model_name = model_name
              self._max_new_tokens = max_new_tokens
              self._model = None  # lazy-loaded on first call

          def generate_text(self, image: PILImage.Image, prompt: str) -> str:
              if self._model is None:
                  import fiftyone.zoo as foz  # deferred import — keeps module importable without GPU
                  self._model = foz.load_zoo_model(self._model_name, max_new_tokens=self._max_new_tokens)
              self._model.config.prompt = prompt  # override per-call prompt (D-14: deterministic by default)
              raw_texts: list[str] = self._model._generate_detections([image])  # bypasses detection output processor
              return raw_texts[0]

      - class MockVlmAdapter:
          """Deterministic mock for CI tests (D-22). No GPU or model download required.
          Matches by checking whether a response key is a substring of the prompt.
          """
          def __init__(self,
                       responses: dict[str, str] | None = None,
                       default_response: str = '{"error_probability": 0.1, "reason": "mock-ok"}',
          ) -> None:
              self._responses: dict[str, str] = responses or {}
              self._default = default_response

          def generate_text(self, image: PILImage.Image, prompt: str) -> str:
              for key, resp in self._responses.items():
                  if key in prompt:
                      return resp
              return self._default
  </action>
